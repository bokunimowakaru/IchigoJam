1 'SENSOR TEST LCD
2 CLS
3 gosub 800
10 A=ANA(2)
20 LC 0,0
30 FOR I=0 TO 7
40 IF A>I*100 ?"#"; ELSE ?"=";
50 NEXT
60 LC 0,1:? A;"  ";
65 gosub 900
70 WAIT 60
80 GOTO 10
800 'LCD INIT
810 poke#700,64,0,2,#C0,57,17,#70,86,#6C,56,12
820 ifi2cw(62,#701,1,#704,5)?"E"
830 wait12
840 ifi2cw(62,#701,1,#709,2)?"E"
900 'LCD OUT
910 ifi2cw(62,#701,1,#702,1)+i2cw(62,#700,1,#900,8)+i2cw(62,#701,1,#703,1)+i2cw(62,#700,1,#920,8)?"E"
920 return
